1666.4.4 Hashes for Named Arguments
167
168# This method returns an array a of n numbers. For any index i, 0 <= i < n,
169# the value of element a[i] is m*i+c. Arguments n, m, and c are passed
170# as keys in a hash, so that it is not necessary to remember their order.
171def sequence(args)
172  # Extract the arguments from the hash.
173  # Note the use of the || operator to specify defaults used
174  # if the hash does not define a key that we are interested in.
175  n = args[:n] || 0
176  m = args[:m] || 1
177  c = args[:c] || 0
178
179  a = []                      # Start with an empty array
180  n.times {|i| a << m*i+c }   # Calculate the value of each array element
181  a                           # Return the array
182end
183---------------------------
184sequence({:n=>3, :m=>5})      # => [0, 5, 10]
185---------------------------
186sequence(:m=>3, :n=>5)        # => [0, 3, 6, 9, 12]
187---------------------------
188# Ruby 1.9 hash syntax
189sequence c:1, m:3, n:5        # => [1, 4, 7, 10, 13]
190---------------------------
191sequence {:m=>3, :n=>5}       # Syntax error!

1936.4.5 Block Arguments
194
195# Generate a sequence of n numbers m*i + c and pass them to the block
196def sequence2(n, m, c) 
197  i = 0
198  while(i < n)         # loop n times
199    yield i*m + c      # pass next element of the sequence to the block
200    i += 1
201  end
202end
203
204# Here is how you might use this version of the method
205sequence2(5, 2, 2) {|x| puts x }  # Print numbers 2, 4, 6, 8, 10
206---------------------------
207def sequence3(n, m, c, &b) # Explicit argument to get block as a Proc
208  i = 0
209  while(i < n)
210    b.call(i*m + c)        # Invoke the Proc with its call method
211    i += 1
212  end
213end
214
215# Note that the block is still passed outside of the parentheses
216sequence3(5, 2, 2) {|x| puts x }
217---------------------------
218# This version expects an explicitly-created Proc object, not a block
219def sequence4(n, m, c, b)  # No ampersand used for argument b
220  i = 0
221  while(i < n)
222    b.call(i*m + c)        # Proc is called explicitly
223    i += 1
224  end
225end
226
227p = Proc.new {|x| puts x }  # Explicitly create a Proc object 
228sequence4(5, 2, 2, p)       # And pass it as an ordinary argument
229---------------------------
230def sequence5(args, &b) # Pass arguments as a hash and follow with a block
231  n, m, c = args[:n], args[:m], args[:c]
232  i = 0
233  while(i < n)
234    b.call(i*m + c)
235    i += 1
236  end
237end
238
239# Expects one or more arguments, followed by a block
240def max(first, *rest, &block) 
241  max = first
242  rest.each {|x| max = x if x > max }
243  block.call(max)
244  max
245end

4526.6.1 Closures and Shared Variables
453
454# Return a pair of lambdas that share access to a local variable.
455def accessor_pair(initialValue=nil)
456  value = initialValue  # A local variable shared by the returned lambdas.
457  getter = lambda { value }          # Return value of local variable.
458  setter = lambda {|x| value = x }   # Change value of local variable.
459  return getter,setter               # Return pair of lambdas to caller.
460end
461
462getX, setX = accessor_pair(0) # Create accessor lambdas for initial value 0.
463puts getX[]        # Prints 0. Note square brackets instead of call.
464setX[10]           # Change the value through one closure.
465puts getX[]        # Prints 10. The change is visible through the other.
466---------------------------

518# This module defines methods and operators for functional programming.
519module Functional
520
521  # Apply this function to each element of the specified Enumerable,
522  # returning an array of results. This is the reverse of Enumerable.map.
523  # Use | as an operator alias. Read "|" as "over" or "applied over".
524  # 
525  # Example:
526  #   a = [[1,2],[3,4]]
527  #   sum = lambda {|x,y| x+y}
528  #   sums = sum|a   # => [3,7]
529  def apply(enum)
530    enum.map &self
531  end
532  alias | apply
533
534  # Use this function to "reduce" an enumerable to a single quantity.
535  # This is the inverse of Enumerable.inject.
536  # Use <= as an operator alias.
537  # Mnemonic: <= looks like a needle for injections
538  # Example:
539  #   data = [1,2,3,4]
540  #   sum = lambda {|x,y| x+y}
541  #   total = sum<=data   # => 10
542  def reduce(enum)
543    enum.inject &self
544  end
545  alias <= reduce
546end

549class Proc; include Functional; end
550class Method; include Functional; end

5586.8.2 Composing Functions
559
560module Functional
561  # Return a new lambda that computes self[f[args]].
562  # Use * as an operator alias for compose.
563  # Examples, using the * alias for this method.
564  # 
565  # f = lambda {|x| x*x }
566  # g = lambda {|x| x+1 }
567  # (f*g)[2]   # => 9
568  # (g*f)[2]   # => 5
569  # 
570  # def polar(x,y)
571  #   [Math.hypot(y,x), Math.atan2(y,x)]
572  # end
573  # def cartesian(magnitude, angle)
574  #   [magnitude*Math.cos(angle), magnitude*Math.sin(angle)]
575  # end
576  # p,c = method :polar, method :cartesian
577  # (c*p)[3,4]  # => [3,4]
578  # 
579  def compose(f)
580    if self.respond_to?(:arity) && self.arity == 1
581      lambda {|*args| self[f[*args]] }
582    else
583      lambda {|*args| self[*f[*args]] }
584    end
585  end
586
587  # * is the natural operator for function composition.
588  alias * compose
589end

6326.8.4 Memoizing Functions
633
634module Functional
635  #
636  # Return a new lambda that caches the results of this function and 
637  # only calls the function when new arguments are supplied.
638  #
639  def memoize
640    cache = {}  # An empty cache. The lambda captures this in its closure.
641    lambda {|*args|
642      # notice that the hash key is the entire array of arguments!
643      unless cache.has_key?(args)  # If no cached result for these args
644        cache[args] = self[*args]  # Compute and cache the result
645      end
646      cache[args]                  # Return result from cache
647    }
648  end
649  # A (probably unnecessary) unary + operator for memoization
650  # Mnemonic: the + operator means "improved"
651  alias +@ memoize        # cached_f = +f
652end

713#
714# Add [] and []= operators to the Symbol class for accessing and setting
715# singleton methods of objects. Read : as "method" and [] as "of".
716# So :m[o] reads "method m of o".
717#
718class Symbol
719  # Return the Method of obj named by this symbol. This may be a singleton
720  # method of obj (such as a class method) or an instance method defined
721  # by obj.class or inherited from a superclass.
722  # Examples:
723  #   creator = :new[Object]  # Class method Object.new
724  #   doubler = :*[2]         # * method of Fixnum 2
725  #
726  def [](obj)
727    obj.method(self)
728  end
729  
730  # Define a singleton method on object o, using Proc or Method f as its body.
731  # This symbol is used as the name of the method.
732  # Examples:
733  #
734  #  :singleton[o] = lambda { puts "this is a singleton method of o" }
735  #  :class_method[String] = lambda { puts "this is a class method" }
736  # 
737  # Note that you can't create instance methods this way. See Module.[]=
738  #
739  def []=(o,f)
740    # We can't use self in the block below, as it is evaluated in the 
741    # context of a different object. So we have to assign self to a variable.
742    sym = self
743    # This is the object we define singleton methods on.
744    eigenclass = (class << o; self end)
745    # define_method is private, so we have to use instance_eval to execute it.
746    eigenclass.instance_eval { define_method(sym, f) }
747  end
748end
