130#
131# This class represents a sequence of numbers characterized by the three
132# parameters from, to, and by. The numbers x in the sequence obey the
133# following two constraints:
134#
135#    from <= x <= to
136#    x = from + n*by, where n is an integer
137# 
138class Sequence
139  # This is an enumerable class; it defines an each iterator below.
140  include Enumerable   # Include the methods of this module in this class
141
142  # The initialize method is special; it is automatically invoked to
143  # initialize newly created instances of the class
144  def initialize(from, to, by)
145    # Just save our parameters into instance variables for later use
146    @from, @to, @by = from, to, by  # Note parallel assignment and @ prefix
147  end
148
149  # This is the iterator required by the Enumerable module
150  def each
151    x = @from       # Start at the starting point
152    while x <= @to  # While we haven't reached the end
153      yield x       # Pass x to the block associated with the iterator
154      x += @by      # Increment x
155    end
156  end
157
158  # Define the length method (following arrays) to return the number of
159  # values in the sequence
160  def length
161    return 0 if @from > @to       # Note if used as a statement modifier 
162    Integer((@to-@from)/@by) + 1  # Compute and return length of sequence
163  end
164
165  # Define another name for the same method.
166  # It is common for methods to have multiple names in Ruby
167  alias size length  # size is now a synonym for length
168
169  # Override the array-access operator to give random access to the sequence
170  def[](index)
171    return nil if index < 0 # Return nil for negative indexes
172    v = @from + index*@by   # Compute the value
173    if v <= @to             # If it is part of the sequence
174      v                     # Return it
175    else                    # Otherwise...
176      nil                   # Return nil
177    end
178  end
179
180  # Override arithmetic operators to return new Sequence objects
181  def *(factor)
182    Sequence.new(@from*factor, @to*factor, @by*factor)
183  end
184
185  def +(offset)
186    Sequence.new(@from+offset, @to+offset, @by)
187  end
188end

2921.4 A Sudoku Solver in Ruby
293
294#
295# This module defines a Sudoku::Puzzle class to represent a 9x9
296# Sudoku puzzle and also defines exception classes raised for 
297# invalid input and over-constrained puzzles. This module also defines 
298# the method Sudoku.solve to solve a puzzle. The solve method uses
299# the Sudoku.scan method, which is also defined here.
300# 
301# Use this module to solve Sudoku puzzles with code like this:
302#
303#  require 'sudoku'
304#  puts Sudoku.solve(Sudoku::Puzzle.new(ARGF.readlines))
305#
306module Sudoku
307
308  #
309  # The Sudoku::Puzzle class represents the state of a 9x9 Sudoku puzzle.
310  # 
311  # Some definitions and terminology used in this implementation: 
312  #
313  # - Each element of a puzzle is called a "cell".
314  # - Rows and columns are numbered from 0 to 8, and the coordinates [0,0]
315  #   refer to the cell in the upper-left corner of the puzzle.
316  # - The nine 3x3 subgrids are known as "boxes" and are also numbered from
317  #   0 to 8, ordered from left to right and top to bottom. The box in
318  #   the upper-left is box 0. The box in the upper-right is box 2. The
319  #   box in the middle is box 4. The box in the lower-right is box 8.
320  # 
321  # Create a new puzzle with Sudoku::Puzzle.new, specifying the initial
322  # state as a string or as an array of strings. The string(s) should use
323  # the characters 1 through 9 for the given values, and '.' for cells
324  # whose value is unspecified. Whitespace in the input is ignored.
325  #
326  # Read and write access to individual cells of the puzzle is through the
327  # [] and []= operators, which expect two-dimensional [row,column] indexing.
328  # These methods use numbers (not characters) 0 to 9 for cell contents.
329  # 0 represents an unknown value.
330  # 
331  # The has_duplicates? predicate returns true if the puzzle is invalid
332  # because any row, column, or box includes the same digit twice.
333  #
334  # The each_unknown method is an iterator that loops through the cells of
335  # the puzzle and invokes the associated block once for each cell whose
336  # value is unknown.
337  #
338  # The possible method returns an array of integers in the range 1..9.
339  # The elements of the array are the only values allowed in the specified
340  # cell. If this array is empty, then the puzzle is over-specified and 
341  # cannot be solved. If the array has only one element, then that element
342  # must be the value for that cell of the puzzle.
343  #
344  class Puzzle
345
346    # These constants are used for translating between the external 
347    # string representation of a puzzle and the internal representation.
348    ASCII = ".123456789"
349    BIN = "\000\001\002\003\004\005\006\007\010\011"
350
351    # This is the initialization method for the class. It is automatically
352    # invoked on new Puzzle instances created with Puzzle.new. Pass the input
353    # puzzle as an array of lines or as a single string. Use ASCII digits 1
354    # to 9 and use the '.' character for unknown cells. Whitespace, 
355    # including newlines, will be stripped.
356    def initialize(lines)
357      if (lines.respond_to? :join)  # If argument looks like an array of lines
358        s = lines.join              # Then join them into a single string
359      else                          # Otherwise, assume we have a string
360        s = lines.dup               # And make a private copy of it
361      end
362
363      # Remove whitespace (including newlines) from the data
364      # The '!' in gsub! indicates that this is a mutator method that
365      # alters the string directly rather than making a copy.
366      s.gsub!(/\s/, "")  # /\s/ is a Regexp that matches any whitespace
367
368      # Raise an exception if the input is the wrong size.
369      # Note that we use unless instead of if, and use it in modifier form.
370      raise Invalid, "Grid is the wrong size" unless s.size == 81
371      
372      # Check for invalid characters, and save the location of the first.
373      # Note that we assign and test the value assigned at the same time.
374      if i = s.index(/[^123456789\.]/)
375        # Include the invalid character in the error message.
376        # Note the Ruby expression inside #{} in string literal.
377        raise Invalid, "Illegal character #{s[i,1]} in puzzle"
378      end
379
380      # The following two lines convert our string of ASCII characters
381      # to an array of integers, using two powerful String methods.
382      # The resulting array is stored in the instance variable @grid
383      # The number 0 is used to represent an unknown value.
384      s.tr!(ASCII, BIN)      # Translate ASCII characters into bytes
385      @grid = s.unpack('c*') # Now unpack the bytes into an array of numbers
386
387      # Make sure that the rows, columns, and boxes have no duplicates.
388      raise Invalid, "Initial puzzle has duplicates" if has_duplicates?
389    end
390
391    # Return the state of the puzzle as a string of 9 lines with 9 
392    # characters (plus newline) each.  
393    def to_s
394      # This method is implemented with a single line of Ruby magic that
395      # reverses the steps in the initialize() method. Writing dense code
396      # like this is probably not good coding style, but it demonstrates
397      # the power and expressiveness of the language.
398      #
399      # Broken down, the line below works like this:
400      # (0..8).collect invokes the code in curly braces 9 times--once
401      # for each row--and collects the return value of that code into an
402      # array. The code in curly braces takes a subarray of the grid
403      # representing a single row and packs its numbers into a string.
404      # The join() method joins the elements of the array into a single
405      # string with newlines between them. Finally, the tr() method
406      # translates the binary string representation into ASCII digits.
407      (0..8).collect{|r| @grid[r*9,9].pack('c9')}.join("\n").tr(BIN,ASCII)
408    end
409
410    # Return a duplicate of this Puzzle object.
411    # This method overrides Object.dup to copy the @grid array.
412    def dup
413      copy = super       # Make a shallow copy by calling Object.dup
414      @grid = @grid.dup  # Make a new copy of the internal data 
415      copy               # Return the copied object
416    end
417
418    # We override the array access operator to allow access to the 
419    # individual cells of a puzzle. Puzzles are two-dimensional,
420    # and must be indexed with row and column coordinates.
421    def [](row, col)
422      # Convert two-dimensional (row,col) coordinates into a one-dimensional
423      # array index and get and return the cell value at that index
424      @grid[row*9 + col]
425    end
426
427    # This method allows the array access operator to be used on the 
428    # lefthand side of an assignment operation. It sets the value of 
429    # the cell at (row, col) to newvalue.
430    def []=(row, col, newvalue)
431      # Raise an exception unless the new value is in the range 0 to 9.
432      unless (0..9).include? newvalue
433        raise Invalid, "illegal cell value" 
434      end
435      # Set the appropriate element of the internal array to the value.
436      @grid[row*9 + col] = newvalue
437    end
438
439    # This array maps from one-dimensional grid index to box number.
440    # It is used in the method below. The name BoxOfIndex begins with a 
441    # capital letter, so this is a constant. Also, the array has been
442    # frozen, so it cannot be modified.
443    BoxOfIndex = [
444      0,0,0,1,1,1,2,2,2,0,0,0,1,1,1,2,2,2,0,0,0,1,1,1,2,2,2,
445      3,3,3,4,4,4,5,5,5,3,3,3,4,4,4,5,5,5,3,3,3,4,4,4,5,5,5,
446      6,6,6,7,7,7,8,8,8,6,6,6,7,7,7,8,8,8,6,6,6,7,7,7,8,8,8
447    ].freeze
448
449    # This method defines a custom looping construct (an "iterator") for
450    # Sudoku puzzles.  For each cell whose value is unknown, this method
451    # passes ("yields") the row number, column number, and box number to the 
452    # block associated with this iterator.
453    def each_unknown
454      0.upto 8 do |row|             # For each row
455        0.upto 8 do |col|           # For each column
456          index = row*9+col         # Cell index for (row,col)
457          next if @grid[index] != 0 # Move on if we know the cell's value 
458          box = BoxOfIndex[index]   # Figure out the box for this cell
459          yield row, col, box       # Invoke the associated block
460        end
461      end
462    end
463
464    # Returns true if any row, column, or box has duplicates.
465    # Otherwise returns false. Duplicates in rows, columns, or boxes are not
466    # allowed in Sudoku, so a return value of true means an invalid puzzle.
467    def has_duplicates?
468      # uniq! returns nil if all the elements in an array are unique.
469      # So if uniq! returns something then the board has duplicates.
470      0.upto(8) {|row| return true if rowdigits(row).uniq! }
471      0.upto(8) {|col| return true if coldigits(col).uniq! }
472      0.upto(8) {|box| return true if boxdigits(box).uniq! }
473      
474      false  # If all the tests have passed, then the board has no duplicates
475    end
476
477    # This array holds a set of all Sudoku digits. Used below.
478    AllDigits = [1, 2, 3, 4, 5, 6, 7, 8, 9].freeze
479
480    # Return an array of all values that could be placed in the cell 
481    # at (row,col) without creating a duplicate in the row, column, or box.
482    # Note that the + operator on arrays does concatenation but that the - 
483    # operator performs a set difference operation.
484    def possible(row, col, box)
485      AllDigits - (rowdigits(row) + coldigits(col) + boxdigits(box))
486    end
487
488    private  # All methods after this line are private to the class
489
490    # Return an array of all known values in the specified row.
491    def rowdigits(row)
492      # Extract the subarray that represents the row and remove all zeros.
493      # Array subtraction is set difference, with duplicate removal.
494      @grid[row*9,9] - [0]
495    end
496
497    # Return an array of all known values in the specified column.
498    def coldigits(col)
499      result = []                # Start with an empty array
500      col.step(80, 9) {|i|       # Loop from col by nines up to 80
501        v = @grid[i]             # Get value of cell at that index
502        result << v if (v != 0)  # Add it to the array if non-zero
503      }
504      result                     # Return the array
505    end
506
507    # Map box number to the index of the upper-left corner of the box.
508    BoxToIndex = [0, 3, 6, 27, 30, 33, 54, 57, 60].freeze
509
510    # Return an array of all the known values in the specified box.
511    def boxdigits(b)
512      # Convert box number to index of upper-left corner of the box.
513      i = BoxToIndex[b]
514      # Return an array of values, with 0 elements removed.
515      [
516        @grid[i],    @grid[i+1],  @grid[i+2],
517        @grid[i+9],  @grid[i+10], @grid[i+11],
518        @grid[i+18], @grid[i+19], @grid[i+20]
519      ] - [0]
520    end
521  end  # This is the end of the Puzzle class
522
523  # An exception of this class indicates invalid input,
524  class Invalid < StandardError
525  end
526
527  # An exception of this class indicates that a puzzle is over-constrained
528  # and that no solution is possible.
529  class Impossible < StandardError
530  end
531
532  #
533  # This method scans a Puzzle, looking for unknown cells that have only
534  # a single possible value. If it finds any, it sets their value. Since
535  # setting a cell alters the possible values for other cells, it 
536  # continues scanning until it has scanned the entire puzzle without 
537  # finding any cells whose value it can set.
538  #
539  # This method returns three values. If it solves the puzzle, all three 
540  # values are nil. Otherwise, the first two values returned are the row and
541  # column of a cell whose value is still unknown. The third value is the
542  # set of values possible at that row and column. This is a minimal set of
543  # possible values: there is no unknown cell in the puzzle that has fewer
544  # possible values. This complex return value enables a useful heuristic 
545  # in the solve() method: that method can guess at values for cells where
546  # the guess is most likely to be correct.
547  # 
548  # This method raises Impossible if it finds a cell for which there are
549  # no possible values. This can happen if the puzzle is over-constrained,
550  # or if the solve() method below has made an incorrect guess.
551  #
552  # This method mutates the specified Puzzle object in place.
553  # If has_duplicates? is false on entry, then it will be false on exit.
554  #
555  def Sudoku.scan(puzzle)
556    unchanged = false  # This is our loop variable
557
558    # Loop until we've scanned the whole board without making a change.
559    until unchanged 
560      unchanged = true      # Assume no cells will be changed this time
561      rmin,cmin,pmin = nil  # Track cell with minimal possible set
562      min = 10              # More than the maximal number of possibilities
563
564      # Loop through cells whose value is unknown.
565      puzzle.each_unknown do |row, col, box|
566        # Find the set of values that could go in this cell
567        p = puzzle.possible(row, col, box)
568        
569        # Branch based on the size of the set p. 
570        # We care about 3 cases: p.size==0, p.size==1, and p.size > 1.
571        case p.size
572        when 0  # No possible values means the puzzle is over-constrained
573          raise Impossible
574        when 1  # We've found a unique value, so set it in the grid
575          puzzle[row,col] = p[0] # Set that position on the grid to the value
576          unchanged = false      # Note that we've made a change
577        else    # For any other number of possibilities
578          # Keep track of the smallest set of possibilities.
579          # But don't bother if we're going to repeat this loop.
580          if unchanged && p.size < min
581            min = p.size                    # Current smallest size
582            rmin, cmin, pmin = row, col, p  # Note parallel assignment
583          end
584        end
585      end
586    end
587      
588    # Return the cell with the minimal set of possibilities.
589    # Note multiple return values.
590    return rmin, cmin, pmin
591  end
592
593  # Solve a Sudoku puzzle using simple logic, if possible, but fall back
594  # on brute-force when necessary. This is a recursive method. It either
595  # returns a solution or raises an exception. The solution is returned
596  # as a new Puzzle object with no unknown cells. This method does not 
597  # modify the Puzzle it is passed. Note that this method cannot detect
598  # an under-constrained puzzle.
599  def Sudoku.solve(puzzle)
600    # Make a private copy of the puzzle that we can modify.
601    puzzle = puzzle.dup
602
603    # Use logic to fill in as much of the puzzle as we can.
604    # This method mutates the puzzle we give it, but always leaves it valid.
605    # It returns a row, a column, and set of possible values at that cell.
606    # Note parallel assignment of these return values to three variables.
607    r,c,p = scan(puzzle)
608
609    # If we solved it with logic, return the solved puzzle.
610    return puzzle if r == nil
611    
612    # Otherwise, try each of the values in p for cell [r,c].
613    # Since we're picking from a set of possible values, the guess leaves
614    # the puzzle in a valid state. The guess will either lead to a solution
615    # or to an impossible puzzle. We'll know we have an impossible
616    # puzzle if a recursive call to scan throws an exception. If this happens
617    # we need to try another guess, or re-raise an exception if we've tried
618    # all the options we've got.
619    p.each do |guess|        # For each value in the set of possible values
620      puzzle[r,c] = guess    # Guess the value
621      
622      begin
623        # Now try (recursively) to solve the modified puzzle.
624        # This recursive invocation will call scan() again to apply logic
625        # to the modified board, and will then guess another cell if needed.
626        # Remember that solve() will either return a valid solution or 
627        # raise an exception.  
628        return solve(puzzle)  # If it returns, we just return the solution
629      rescue Impossible
630        next                  # If it raises an exception, try the next guess
631      end
632    end
633
634    # If we get here, then none of our guesses worked out
635    # so we must have guessed wrong sometime earlier.
636    raise Impossible
637  end
638end
