 344# This method expects a block. It generates n values of the form
 345# m*i + c, for i from 0..n-1, and yields them, one at a time, 
 346# to the associated block.
 347def sequence(n, m, c)
 348  i = 0
 349  while(i < n)      # Loop n times
 350    yield m*i + c   # Invoke the block, and pass a value to it
 351    i += 1          # Increment i each time
 352  end
 353end

 355# Here is an invocation of that method, with a block.
 356# It prints the values 1, 6, and 11

 358---------------------------
 359# Generate n points evenly spaced around the circumference of a 
 360# circle of radius r centered at (0,0). Yield the x and y coordinates
 361# of each point to the associated block.
 362def circle(r,n)
 363  n.times do |i|    # Notice that this method is implemented with a block
 364    angle = Math::PI * 2 * i / n
 365    yield r*Math.cos(angle), r*Math.sin(angle)
 366  end
 367end

 373# Return an array with n elements of the form m*i+c
 374# If a block is given, also yield each element to the block
 375def sequence(n, m, c)
 376  i, s = 0, []                  # Initialize variables
 377  while(i < n)                  # Loop n times
 378    y = m*i + c                 # Compute value

 451# Call the each method of each collection in turn.
 452# This is not a parallel iteration and does not require enumerators.
 453def sequence(*enumerables, &block)
 454  enumerables.each do |enumerable|
 455    enumerable.each(&block)
 456  end
 457end
 458
 459# Iterate the specified collections, interleaving their elements.
 460# This can't be done efficiently without external iterators.
 461# Note the use of the uncommon else clause in begin/rescue.
 462def interleave(*enumerables)
 463  # Convert enumerable collections to an array of enumerators.
 464  enumerators = enumerables.map {|e| e.to_enum }
 465  # Loop until we don't have any more enumerators.
 466  until enumerators.empty?
 467    begin
 468      e = enumerators.shift   # Take the first enumerator
 469      yield e.next            # Get its next and pass to the block
 470    rescue StopIteration      # If no more elements, do nothing
 471    else                      # If no exception occurred
 472      enumerators << e        # Put the enumerator back
 473    end
 474  end
 475end
 476
 477# Iterate the specified collections, yielding tuples of values,
 478# one value from each of the collections. See also Enumerable.zip.
 479def bundle(*enumerables)
 480  enumerators = enumerables.map {|e| e.to_enum }
 481  loop { yield enumerators.map {|e| e.next} }
 482end
 483
 484# Examples of how these iterator methods work
 485a,b,c = [1,2,3], 4..6, 'a'..'e'
 486sequence(a,b,c) {|x| print x}   # prints "123456abcde"
 487interleave(a,b,c) {|x| print x} # prints "14a25b36cde"
 488bundle(a,b,c) {|x| print x}     # '[1, 4, "a"][2, 5, "b"][3, 6, "c"]'

 996def fibonacci_generator(x0,y0)   # Base the sequence on x0,y0

1002    end
1003  end

1040class Generator
1041  def initialize(enumerable)
1042    @enumerable = enumerable  # Remember the enumerable object
1043    create_fiber              # Create a fiber to enumerate it
1044  end
1045
1046  def next                    # Return the next element
1047    @fiber.resume             # by resuming the fiber
1048  end
1049
1050  def rewind                  # Start the enumeration over
1051    create_fiber              # by creating a new fiber
1052  end
1053
1054  private
1055  def create_fiber            # Create the fiber that does the enumeration
1056    @fiber = Fiber.new do     # Create a new fiber
1057      @enumerable.each do |x| # Use the each method
1058        Fiber.yield(x)        # But pause during enumeration to return values
1059      end              
1060      raise StopIteration     # Raise this when we're out of values
1061    end
1062  end
1063end
