190module Stats                          # A module
191  class Dataset                       # A class in the module
192    def initialize(filename)          # A method in the class
193      IO.foreach(filename) do |line|  # A block in the method
194        if line[0,1] == "#"           # An if statement in the block
195          next                        # A simple statement in the if
196        end                           # End the if body
197      end                             # End the block
198    end                               # End the method body
199  end                                 # End the class body
