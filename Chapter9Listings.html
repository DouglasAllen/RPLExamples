 2689.2.3 Regular Expression Syntax
 269
 270# Literal characters
 271/ruby/             # Match "ruby". Most characters simply match themselves.
 272/&#xA5;/                # Matches Yen sign. Multibyte characters are suported
 273                   # in Ruby 1.9 and Ruby 1.8.
 274
 275# Character classes
 276/[Rr]uby/          # Match "Ruby" or "ruby"
 277/rub[ye]/          # Match "ruby" or "rube"
 278/[aeiou]/          # Match any one lowercase vowel
 279/[0-9]/            # Match any digit; same as /[0123456789]/
 280/[a-z]/            # Match any lowercase ASCII letter
 281/[A-Z]/            # Match any uppercase ASCII letter
 282/[a-zA-Z0-9]/      # Match any of the above
 283/[^aeiou]/         # Match anything other than a lowercase vowel
 284/[^0-9]            # Match anything other than a digit
 285
 286# Special character classes
 287/./                # Match any character except newline
 288/./m               # In multiline mode . matches newline, too
 289/\d/               # Match a digit /[0-9]/
 290/\D/               # Match a nondigit: /[^0-9]/
 291/\s/               # Match a whitespace character: /[ \t\r\n\f]/
 292/\S/               # Match nonwhitespace: /[^ \t\r\n\f]/
 293/\w/               # Match a single word character: /[A-Za-z0-9_]/
 294/\W/               # Match a nonword character: /[^A-Za-z0-9_]/
 295
 296# Repetition
 297/ruby?/            # Match "rub" or "ruby": the y is optional
 298/ruby*/            # Match "rub" plus 0 or more ys
 299/ruby+/            # Match "rub" plus 1 or more ys
 300/\d{3}/            # Match exactly 3 digits
 301/\d{3,}/           # Match 3 or more digits
 302/\d{3,5}/          # Match 3, 4, or 5 digits
 303
 304# Nongreedy repetition: match the smallest number of repetitions
 305/<.*>/             # Greedy repetition: matches "<ruby>perl>"
 306/<.*?>/            # Nongreedy: matches "<ruby>" in "<ruby>perl>" 
 307                   # Also nongreedy: ??, +?, and {n,m}?
 308
 309# Grouping with parentheses
 310/\D\d+/            # No group: + repeats \d
 311/(\D\d)+/          # Grouped: + repeats \D\d pair
 312/([Rr]uby(, )?)+/  # Match "Ruby", "Ruby, ruby, ruby", etc.
 313
 314# Backreferences: matching a previously matched group again
 315/([Rr])uby&\1ails/ # Match ruby&rails or Ruby&Rails
 316/(['"])[^\1]*\1/   # Single or double-quoted string
 317                   #   \1 matches whatever the 1st group matched
 318                   #   \2 matches whatever the 2nd group matched, etc.
 319
 320# Named groups and backreferences in Ruby 1.9: match a 4-letter palindrome
 321/(?<first>\w)(?<second>\w)\k<second>\k<first>/
 322/(?'first'\w)(?'second'\w)\k'second'\k'first'/ # Alternate syntax
 323

 325/ruby|rube/        # Match "ruby" or "rube"
 326/rub(y|le))/       # Match "ruby" or "ruble"
 327/ruby(!+|\?)/      # "ruby" followed by one or more ! or one ?
 328
 329# Anchors: specifying match position
 330/^Ruby/            # Match "Ruby" at the start of a string or internal line
 331/Ruby$/            # Match "Ruby" at the end of a string or line
 332/\ARuby/           # Match "Ruby" at the start of a string
 333/Ruby\Z/           # Match "Ruby" at the end of a string
 334/\bRuby\b/         # Match "Ruby" at a word boundary
 335/\brub\B/          # \B is nonword boundary:
 336                   #   match "rub" in "rube" and "ruby" but not alone
 337/Ruby(?=!)/        # Match "Ruby", if followed by an exclamation point
 338/Ruby(?!!)/        # Match "Ruby", if not followed by an exclamation point

 342/R(?i)uby/         # Case-insensitive while matching "uby"
