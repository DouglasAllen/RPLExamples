2834.6.9.1 Boolean flip-flops
284
285(1..10).each {|x| print x if x==3..x==5 }
286---------------------------
287# Prints "3". Flips and flops back when x==3
288(1..10).each {|x| print x if x==3..x>=3 }  
289# Prints "34". Flips when x == 3 and flops when x==4
290(1..10).each {|x| print x if x==3...x>=3 } # Prints "34"
291---------------------------
292ARGF.each do |line|   # For each line of standard in or of named files
293  print line if line=~/TODO/..line=~/^$/ # Print lines when flip-flop is true
294end
295---------------------------
296$state = false            # Global storage for flip-flop state
297def flipflop(x)           # Test value of x against flip-flop
298  if !$state              # If saved state is false
299    result = (x == 3)     # Result is value of lefthand operand
300    if result             # If that result is true
301      $state = !(x == 5)  # Then saved state is not of the righthand operand
302    end
303    result                # Return result
304  else                    # Otherwise, if saved state is true
305    $state = !(x == 5)    # Then save the inverse of the righthand operand
306    true                  # And return true without testing lefthand
307  end
308end
309---------------------------
310(1..10).each {|x| print x if flipflop(x) }
311---------------------------
312$state2 = false
313def flipflop2(x)
314  if !$state2
315    $state2 = (x == 3)
316  else
317    $state2 = !(x >= 3)
318    true
319  end
320end
321
322# Now try it out
323(1..10).each {|x| print x if x==3...x>=3 }  # Prints "34" 
324(1..10).each {|x| print x if flipflop2(x) } # Prints "34"
